// Current prompt state
let currentPrompt = {
  characters: [],
  clothing: '',
  props: [],
  decorations: [],
  effects: [],
  typography: ''
};

// Helpers
function getRandomItem(array) { return array[Math.floor(Math.random() * array.length)]; }
function getRandomItems(array, count) {
  const result = [];
  for(let i=0;i<count;i++) result.push(getRandomItem(array));
  return result;
}

// Generate full prompt
function generatePrompt() {
  const selectedChars = Array.from(document.querySelectorAll('.charOption:checked')).map(c => c.value);
  currentPrompt.characters = selectedChars.map(cat => getRandomItem(halloweenElements.Characters[cat]));
  currentPrompt.clothing = document.getElementById('includeClothing').checked ? getRandomItem(halloweenElements.ClothingAndAccessories) : '';
  const propsCount = parseInt(document.getElementById('propsCount').value);
  currentPrompt.props = propsCount > 0 ? getRandomItems(halloweenElements.Objects, propsCount) : [];
  const decorCount = parseInt(document.getElementById('decorCount').value);
  currentPrompt.decorations = decorCount > 0 ? getRandomItems(halloweenElements.Decorations, decorCount) : [];
  const effectsCount = parseInt(document.getElementById('effectsCount').value);
  currentPrompt.effects = effectsCount > 0 ? getRandomItems(halloweenElements.AtmosphereAndEffects, effectsCount) : [];
  currentPrompt.typography = document.getElementById('includeTypography').checked ? getRandomItem(halloweenElements.Typography) : '';
  displayPrompt();
}

// Display the current prompt
function displayPrompt() {
  const promptDiv = document.getElementById('prompt');
  const parts = [];
  if(currentPrompt.characters.length) parts.push("Characters: " + currentPrompt.characters.join(', '));
  if(currentPrompt.clothing) parts.push("Clothing/Accessories: " + currentPrompt.clothing);
  if(currentPrompt.props.length) parts.push("Props/Objects: " + currentPrompt.props.join(', '));
  if(currentPrompt.decorations.length) parts.push("Decorations: " + currentPrompt.decorations.join(', '));
  if(currentPrompt.effects.length) parts.push("Atmosphere/Effects: " + currentPrompt.effects.join(', '));
  if(currentPrompt.typography) parts.push("Typography/Text: " + currentPrompt.typography);
  promptDiv.innerText = "Create a detailed illustration featuring:\n" + parts.join("\n");
  promptDiv.style.height = 'auto';
}

// Copy prompt
function copyPrompt() {
  navigator.clipboard.writeText(document.getElementById('prompt').innerText)
    .then(()=>alert("Prompt copied to clipboard!"))
    .catch(err=>alert("Failed to copy: "+err));
}
